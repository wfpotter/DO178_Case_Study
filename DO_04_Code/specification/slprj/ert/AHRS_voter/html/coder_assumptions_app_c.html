18/* combine arguments (after expanding arguments) */



19#define CA_EXP_CONCAT(a,b) CA_CONCAT(a,b)



20#define CA_CONCAT(a,b) a ## b

22/* Compile time static assert for checking fundamental assumptions such 



23 * as sizeof values for specific data types. */



24#define CA_STATIC_ASSERT(expr, msg) typedef char CA_EXP_CONCAT(assert_failed__, msg) [(expr) ? (+1) : (-1)]



25



26#include "coder_assumptions_hwimpl.h"



27



28#ifdef HOST_WORD_ADDRESSABLE_TESTING



29   /* MemUnit_T can be any of uint8_T, uint16_T, uint32_T



30    * Just check it is consistent with MEM_UNIT_BYTES */



31   CA_STATIC_ASSERT(sizeof(MemUnit_T) == MEM_UNIT_BYTES, INVALID_MEMUNIT_T);



32#else



33   /* Verify MemUnit_T: sizeof must be 1 */



34   CA_STATIC_ASSERT(sizeof(MemUnit_T) == 1, MEMUNIT_T_NOT_SIZEOF_1);



35   /* Verify MEM_UNIT_BYTES: must match the size of the largest integer that has sizeof == 1 



36    * Only word addressable targets up to 32-bits are supported. */



37   CA_STATIC_ASSERT((sizeof(uint32_T) == 1) ? MEM_UNIT_BYTES == 4 : 



38                 (sizeof(uint16_T) == 1) ? MEM_UNIT_BYTES == 2 :



39                 (sizeof(uint8_T) == 1) ? MEM_UNIT_BYTES == 1 : 0, INVALID_MEM_UNIT_BYTES);



40#endif

47static XIL_DATA_STREAM_ERROR_CODE sendResult(const CA_TestResult result, 



48                                             const MemUnit_T resultType, 



49                                             const MemUnit_T actual, 



50                                             const MemUnit_T assumed) {



51   XIL_DATA_STREAM_ERROR_CODE errorCode;



52   MemUnit_T tmp;



53   /* send result type */



54   errorCode = coderAssumptionsWriteData(&resultType, sizeof(MemUnit_T));   



55   if (errorCode) return errorCode;



56   



57   /* send result status */



58   tmp = (MemUnit_T) result.status;



59   errorCode = coderAssumptionsWriteData(&tmp, sizeof(MemUnit_T));   



60   if (errorCode) return errorCode;



61



62   /* send result msg id */



63   tmp = (MemUnit_T) result.msg;



64   errorCode = coderAssumptionsWriteData(&tmp, sizeof(MemUnit_T));   



65   if (errorCode) return errorCode;



66   



67   /* send actual */



68   errorCode = coderAssumptionsWriteData(&actual, sizeof(MemUnit_T));   



69   if (errorCode) return errorCode;



70   



71   /* send assumed */



72   errorCode = coderAssumptionsWriteData(&assumed, sizeof(MemUnit_T));



73   if (errorCode) return errorCode;



74   



75   /* final return */



76   return errorCode;



77}

93/* portable word sizes specified checks */



94static XIL_DATA_STREAM_ERROR_CODE sendPWS(void) {



95    XIL_DATA_STREAM_ERROR_CODE errorCode; 



96    MemUnit_T commandId;  



97   



98    /* send CA_PWS */



99    commandId = CA_PWS;



100    errorCode = coderAssumptionsWriteData(&commandId, sizeof(MemUnit_T));



101    if (errorCode) return errorCode;



102



103    /* PWS enabled in build */



104    commandId = CA_PWS_ENABLED;



105    errorCode = coderAssumptionsWriteData(&commandId, sizeof(MemUnit_T));



106    if (errorCode) return errorCode;



107



108    pwsEnabled = (MemUnit_T) CA_Actual_HWImpl.portableWordSizes;



109    errorCode = coderAssumptionsWriteData(&pwsEnabled, sizeof(MemUnit_T));



110    if (errorCode) return errorCode;



111



112    /* extended PWS checks */



113    if (pwsEnabled) {



114       /* config setting mismatch */



115       errorCode = sendResult(CA_PortableWordSizes_Results.configSetting, 



116                                    (MemUnit_T) CA_PWS_CONFIG_SETTING,  



117                                    (MemUnit_T) CA_Actual_HWImpl.portableWordSizes, 



118                                    (MemUnit_T) CA_Expected_HWImpl.portableWordSizes);



119       if (errorCode) return errorCode; 



120    }



121



122



123    errorCode = coderAssumptionsSendWriteBuffer();  



124    if (errorCode) return errorCode;



125



126    return errorCode;



127}

296/* hardware implementation, checks unrelated to sizes */



297static XIL_DATA_STREAM_ERROR_CODE sendHardwareChecks(void){



298    XIL_DATA_STREAM_ERROR_CODE errorCode = XIL_DATA_STREAM_SUCCESS;



299    MemUnit_T commandId;



300



301    commandId = CA_CHECKS_1;



302    errorCode = coderAssumptionsWriteData(&commandId, sizeof(MemUnit_T));



303    if (errorCode) return errorCode;



304



305   /* memory at startup */



306   errorCode = sendResult(CA_HWImpl_Results.memoryAtStartup,



307         (MemUnit_T) CA_HW_IMPL_MEMORY_STARTUP,



308         (MemUnit_T) CA_Actual_HWImpl.memoryAtStartup,



309         (MemUnit_T) CA_Expected_HWImpl.memoryAtStartup);



310   if(errorCode) return errorCode;



311   



312   /* denormal flush to zero */



313   errorCode = sendResult(CA_HWImpl_Results.denormalFlushToZero,



314         (MemUnit_T) CA_HW_IMPL_DENORMAL_FTZ,



315         (MemUnit_T) CA_Actual_HWImpl.denormalFlushToZero,



316         (MemUnit_T) CA_Expected_HWImpl.denormalFlushToZero);



317   if(errorCode) return errorCode;



318   



319   /* denormal as zero */



320   errorCode = sendResult(CA_HWImpl_Results.denormalAsZero,



321         (MemUnit_T) CA_HW_IMPL_DENORMAL_DAZ,



322         (MemUnit_T) CA_Actual_HWImpl.denormalAsZero,



323         (MemUnit_T) CA_Expected_HWImpl.denormalAsZero);



324   if(errorCode) return errorCode;



325   /* transmit message */



326   errorCode = coderAssumptionsSendWriteBuffer();



327   if (errorCode) return errorCode;



328



329   return errorCode;



330}

332static XIL_DATA_STREAM_ERROR_CODE runChecks(void) {



333    XIL_DATA_STREAM_ERROR_CODE errorCode;



334    MemUnit_T endTransmission = CA_END_TRANSMIT;



335



336     /* first, run the coder assumptions checks */



337    CA_Run_Tests();



338



339    /* now send the results to the host */



340    errorCode = transmitMemUnitBytes();



341    if (errorCode) return errorCode;



342



343    errorCode = sendPWS();



344    if (errorCode) return errorCode;



345



346    errorCode = sendHWImpl1();



347    if (errorCode) return errorCode;



348    errorCode = sendHWImpl2();



349    if (errorCode) return errorCode;



350    errorCode = sendHWImpl3();



351    if (errorCode) return errorCode;



352    errorCode = sendHWImpl4();



353    if (errorCode) return errorCode;



354



355    /* send results of miscellaneous checks */



356    errorCode = sendHardwareChecks();



357    if (errorCode) return errorCode;



358



359    /* let host know all tests have completed */



360    errorCode = coderAssumptionsWriteData(&endTransmission, sizeof(MemUnit_T));



361    if (errorCode) return errorCode;



362    errorCode = coderAssumptionsSendWriteBuffer();



363    



364    /* final return */



365    return errorCode;



366}
