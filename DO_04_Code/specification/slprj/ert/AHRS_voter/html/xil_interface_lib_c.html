50static void getNextXILIOData(void) {



51   /* increment xilIOData ptr if we have not reached the end */



52   if (xilIODataPtr->memUnitLength!=0) {



53      xilIODataPtr++;



54   }

57static XIL_PROCESSDATA_ERROR_CODE processHostToTargetData(uint32_T fcnId,



58        XIL_COMMAND_TYPE_ENUM command,



59        uint32_T commandIdx)  {



60    



61    /* this function receives some arbitrary data size that needs to be 



62     * into the appropriate memory addresses. Since it is not guaranteed 



63     * that we will copy all the required data in one go, we need to keep 



64     * track of which address (and size) in xilIODataPtr we last copied 



65     * to so we can continue copying the remaining data the next time we 



66     * receive them.



67     */



68    static MemUnit_T * uDataAddress = NULL;



69    static uint32_T uDataLength = 0;



70    int processingComplete = 0;



71        



72	/* initialize uDataAddress and uDataLength if necessary*/



73    if (uDataAddress == NULL) {



74		/* initialise xilIODataPtr before beginning to process data */



75		if (xilGetHostToTargetData(fcnId, command, commandIdx, &xilIODataPtr)) {



76			/* no udata processing to be done - we're complete */



77			processingComplete = 1;			



78			return XIL_PROCESSDATA_SUCCESS;



79		}



80



81        uDataAddress = xilIODataPtr->address;



82        uDataLength = xilIODataPtr->memUnitLength;



83    }



84    



85    while(!(processingComplete)) {



86        xilReadData(uDataAddress, uDataLength);



87        getNextXILIOData();



88            



89        if (xilIODataPtr->memUnitLength == 0) {



90            /* done processing all xilIOData */                



91            processingComplete = 1;



92            uDataAddress = NULL;



93        } else {



94            /* reset our pointer to next xilIOData */



95            uDataAddress = xilIODataPtr->address;



96            uDataLength = xilIODataPtr->memUnitLength;



97        }



98    }



99



100    return XIL_PROCESSDATA_SUCCESS;



101}

103static XIL_PROCESSDATA_ERROR_CODE processTargetToHostData(uint32_T fcnId,



104        MemUnit_T responseId,



105        XIL_COMMAND_TYPE_ENUM command,



106        uint32_T commandIdx)  {



107        



108    int moreXILData = 1;



109    



110    /* initialise before beginning to process data */    



111    if (xilGetTargetToHostData(fcnId, command, commandIdx, &xilIODataPtr)) {



112        /* no ydata processing to be done - we're complete */        



113        return XIL_PROCESSDATA_SUCCESS;



114    }



115    /* send response id */



116    if (xilWriteData(&responseId, sizeof(responseId)) != XIL_DATA_STREAM_SUCCESS) {        



117        return XIL_PROCESSDATA_DATA_STREAM_ERROR;



118    }



119    



120    /* send server function id if this is a caller function */



121    if (responseId == XIL_RESPONSE_CS_REQUEST_SERVICE) {



122        if (xilWriteData((MemUnit_T *) &serverFcnId, sizeof(serverFcnId)) != XIL_DATA_STREAM_SUCCESS) {



123            return XIL_PROCESSDATA_DATA_STREAM_ERROR;



124        }



125    }



126    



127    while(moreXILData) {



128        if (xilWriteData(xilIODataPtr->address, xilIODataPtr->memUnitLength) != XIL_DATA_STREAM_SUCCESS) {            



129            return XIL_PROCESSDATA_DATA_STREAM_ERROR;



130        }



131        /* get next xilIOData */



132        getNextXILIOData();



133        if (xilIODataPtr->memUnitLength == 0) {



134            moreXILData = 0;



135        }



136    }



137    return XIL_PROCESSDATA_SUCCESS;



138}

140static XIL_INTERFACE_LIB_ERROR_CODE finalizeCommandResponse(XIL_RESPONSE_ERROR_ID errorId, MemUnit_T isEndOfXILStep)  {



141



142#ifdef CODE_INSTRUMENTATION_ENABLED



143    /* flush the output stream



144     * before beginning next command */



145    if (codeInstrDataFlush() != XIL_DATA_STREAM_SUCCESS) {



146        return XIL_INTERFACE_LIB_ERROR;



147    }



148#endif



149



150   /* only send the error id if necessary */



151   if (errorId != XIL_RESPONSE_ERROR_SUCCESS) {



152      /* send response id code */



153      MemUnit_T memUnitData = XIL_RESPONSE_ERROR;



154      if (xilWriteData(&memUnitData, sizeof(memUnitData)) != XIL_DATA_STREAM_SUCCESS) {



155        return XIL_INTERFACE_LIB_ERROR;      



156      }



157      /* send error id */



158	  memUnitData = (MemUnit_T) errorId;



159      if (xilWriteData(&memUnitData, sizeof(memUnitData)) != XIL_DATA_STREAM_SUCCESS) {



160         return XIL_INTERFACE_LIB_ERROR;      



161      }	  



162   }



163   /* flush the output stream 



164    * before beginning next command */



165   if (xilDataFlush(isEndOfXILStep) != XIL_DATA_STREAM_SUCCESS) {



166      return XIL_INTERFACE_LIB_ERROR;      



167   }   



168   return XIL_INTERFACE_LIB_SUCCESS;



169}

171/* Given current state and command, compute the next state for handleReceive */



172static XIL_HANDLE_RECV_STATE computeNextState(XIL_HANDLE_RECV_STATE state, XIL_COMMAND_TYPE_ENUM command) {



173    



174    XIL_HANDLE_RECV_STATE nextState;



175    



176    switch(state) {



177        case XIL_READ_COMMAND_CODE:



178            switch(command) {



179                case XIL_INIT_COMMAND:                    



180                case XIL_CONST_OUTPUT_COMMAND:



181                case XIL_INITIALIZE_COMMAND:



182                case XIL_SYSTEM_INITIALIZE_COMMAND:



183                case XIL_TERMINATE_COMMAND:



184                case XIL_PROCESS_PARAMS_COMMAND:



185                case XIL_ENABLE_COMMAND:



186                case XIL_DISABLE_COMMAND:



187                case XIL_CLIENT_SERVER_COMMAND:



188                case XIL_UPDATE_COMMAND:



189                case XIL_OUTPUT_COMMAND:



190                case XIL_SYSTEM_RESET_COMMAND:



191                case XIL_PAUSE_COMMAND:



192                    nextState = XIL_READ_FCNID;



193                    break;



194                case XIL_SHUTDOWN_COMMAND:



195                    nextState = XIL_READ_FCNID;



196                    break;



197                default:



198                    nextState = XIL_INVALID_STATE;



199                    break;



200            }



201            break;



202        case XIL_READ_FCNID:



203            switch(command) {



204                case XIL_INIT_COMMAND:



205                case XIL_PAUSE_COMMAND:



206                    nextState = XIL_CALL_XILINTERFACE;



207                    break;



208                case XIL_CONST_OUTPUT_COMMAND:



209                    nextState = XIL_PROCESS_TARGET_TO_HOST_DATA;



210                    break;



211                case XIL_INITIALIZE_COMMAND:



212                case XIL_SYSTEM_INITIALIZE_COMMAND:



213                case XIL_TERMINATE_COMMAND:



214                case XIL_PROCESS_PARAMS_COMMAND:



215                case XIL_SYSTEM_RESET_COMMAND:



216                    nextState = XIL_PROCESS_HOST_TO_TARGET_DATA;



217                    break;



218                case XIL_ENABLE_COMMAND:



219                case XIL_DISABLE_COMMAND:



220                case XIL_CLIENT_SERVER_COMMAND:  



221                case XIL_UPDATE_COMMAND:



222                case XIL_OUTPUT_COMMAND:



223                    nextState = XIL_READ_XILTID;



224                    break;



225                default:



226                    nextState = XIL_INVALID_STATE;



227                    break;



228            }



229            break;            



230        case XIL_READ_XILTID:



231            switch(command) {



232                case XIL_ENABLE_COMMAND:



233                case XIL_DISABLE_COMMAND:



234                case XIL_CLIENT_SERVER_COMMAND:



235                case XIL_OUTPUT_COMMAND:



236                case XIL_UPDATE_COMMAND:



237                    nextState = XIL_PROCESS_HOST_TO_TARGET_DATA;



238                    break;



239                default:



240                    nextState = XIL_INVALID_STATE;



241                    break;



242            }



243            break;



244        case XIL_PROCESS_HOST_TO_TARGET_DATA:



245            switch(command) {



246                case XIL_INITIALIZE_COMMAND:



247                case XIL_SYSTEM_INITIALIZE_COMMAND:                



248                case XIL_ENABLE_COMMAND:



249                case XIL_DISABLE_COMMAND:



250                case XIL_TERMINATE_COMMAND:



251                case XIL_PROCESS_PARAMS_COMMAND:



252                case XIL_OUTPUT_COMMAND:



253                case XIL_UPDATE_COMMAND:



254                case XIL_SYSTEM_RESET_COMMAND:



255                case XIL_PAUSE_COMMAND:



256                    nextState = XIL_CALL_XILINTERFACE;



257                    break;



258                case XIL_CLIENT_SERVER_COMMAND:



259                    nextState = XIL_COMPLETE;



260                    break;



261                default:



262                    nextState = XIL_INVALID_STATE;



263                    break;



264            }



265            break;



266        case XIL_CALL_XILINTERFACE:



267            switch(command) {



268                case XIL_UPDATE_COMMAND:



269                case XIL_INIT_COMMAND:



270                case XIL_PROCESS_PARAMS_COMMAND:



271                    nextState = XIL_COMPLETE;



272                    break;



273                case XIL_INITIALIZE_COMMAND:



274                case XIL_SYSTEM_INITIALIZE_COMMAND:



275                case XIL_CONST_OUTPUT_COMMAND:



276                case XIL_ENABLE_COMMAND:



277                case XIL_DISABLE_COMMAND:                



278                case XIL_OUTPUT_COMMAND:



279                case XIL_SYSTEM_RESET_COMMAND:



280                case XIL_PAUSE_COMMAND:



281                case XIL_TERMINATE_COMMAND:   



282                    nextState = XIL_PROCESS_TARGET_TO_HOST_DATA;



283                    break;



284                default:



285                    nextState = XIL_INVALID_STATE;



286                    break;



287            }



288            break;



289        case XIL_PROCESS_TARGET_TO_HOST_DATA:



290            /* done */



291            nextState = XIL_COMPLETE;



292            break;



293        default:



294            nextState = XIL_INVALID_STATE;



295            break;



296    }



297    return nextState;



298}

316static void xilReadCommandCode(void) {



317    MemUnit_T commandCode;    



318    if (cc.state==XIL_READ_COMMAND_CODE) {



319        /* read commandCode */



320



321        xilReadData(&(commandCode), XIL_COMMAND_TYPE_SIZE);



322



323        /* cast from the MemUnit commandCode to the enumeration commandCode */



324        cc.command = (XIL_COMMAND_TYPE_ENUM) commandCode;



325            



326        if (cc.command == XIL_SHUTDOWN_COMMAND) {



327			/* Terminate this process when XIL simulation is complete */



328            isShutdownCommand = 1;             



329            if (finalizeCommandResponse(XIL_RESPONSE_ERROR_TERMINATE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



330                XIL_RUN_EXIT_ERROR;



331            }



332            else {



333                XIL_RUN_EXIT;



334            }



335        } else {



336            /* compute next state */



337            cc.state = computeNextState(cc.state, cc.command);



338            XIL_EXIT_ERROR_IF_INVALID_STATE;



339        }



340    }



341}

343void xilProcessMsg(void) { 



344    /* read commandCode */



345    xilReadCommandCode();



346    



347    if (cc.state==XIL_READ_FCNID) {



348        /* read fcnId */



349#ifdef HOST_BIG_ENDIAN_PROTOCOL_TESTING



350        xilReadDataAndSwapBytes(((MemUnit_T*)&(cc.fcnId)), XIL_COMMAND_FCNID_SIZE);



351#else



352        xilReadData(((MemUnit_T*)&(cc.fcnId)), XIL_COMMAND_FCNID_SIZE);



353#endif /*HOST_BIG_ENDIAN_PROTOCOL_TESTING*/



354        



355



356            



357        /* compute next state */



358        cc.state = computeNextState(cc.state, cc.command);



359        XIL_EXIT_ERROR_IF_INVALID_STATE;



360    }



361        



362        



363    if (cc.state==XIL_READ_XILTID) {



364        /* read commandIdx */



365        



366#ifdef HOST_BIG_ENDIAN_PROTOCOL_TESTING



367        xilReadDataAndSwapBytes(((MemUnit_T*)&(cc.commandIdx)), XIL_COMMAND_XILTID_SIZE);



368#else



369        xilReadData(((MemUnit_T*)&(cc.commandIdx)), XIL_COMMAND_XILTID_SIZE);



370#endif /*HOST_BIG_ENDIAN_PROTOCOL_TESTING*/



371



372        /* compute next state */



373        cc.state = computeNextState(cc.state, cc.command);



374            XIL_EXIT_ERROR_IF_INVALID_STATE;



375    }



376        



377    if (cc.state == XIL_PROCESS_HOST_TO_TARGET_DATA) {



378        /* process inputs if required by command */



379        XIL_PROCESSDATA_ERROR_CODE processDataError;



380        /* process data from host */



381        processDataError = processHostToTargetData(cc.fcnId, cc.command, cc.commandIdx);



382        if (processDataError == XIL_PROCESSDATA_DATA_STREAM_ERROR) {



383            XIL_RUN_EXIT_ERROR;



384        } else if (processDataError != XIL_PROCESSDATA_SUCCESS) {



385            if (finalizeCommandResponse(XIL_RESPONSE_ERROR_PROCESS_UDATA, 1) != XIL_INTERFACE_LIB_SUCCESS) {



386                XIL_RUN_EXIT_ERROR;



387            }



388            else {



389                XIL_RUN_EXIT;



390            }



391        }



392            



393        /* compute next state */



394        cc.state = computeNextState(cc.state, cc.command);



395            XIL_EXIT_ERROR_IF_INVALID_STATE



396    }



397        



398    if (cc.state == XIL_CALL_XILINTERFACE) {



399        /* call xilInterface function */



400        switch(cc.command) {



401#ifndef OUT_OF_PROCESS_EXEC



402            case XIL_INIT_COMMAND:



403            {



404#if CODER_ASSUMPTIONS_ENABLED == 1



405                /* run validation of coder assumptions */



406                if (coderAssumptionsRun() != XIL_INTERFACE_LIB_SUCCESS) {



407                    XIL_RUN_EXIT_ERROR;



408                } 



409#endif



410                if (xilGetDataTypeInfo() != XIL_INTERFACE_SUCCESS) {



411                    XIL_RUN_EXIT_ERROR;



412                }



413                break;



414            }



415            case XIL_CONST_OUTPUT_COMMAND:



416                /* no function */



417                break;



418            case XIL_PROCESS_PARAMS_COMMAND:



419                if (xilProcessParams(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



420                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_PROCESS_PARAMS, 1) != XIL_INTERFACE_LIB_SUCCESS) {



421                        XIL_RUN_EXIT_ERROR;



422                    }



423                    else {



424                        XIL_RUN_EXIT;



425                    }



426                }



427                break;



428            case XIL_INITIALIZE_COMMAND:



429                if (xilInitialize(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



430                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_INITIALIZE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



431                        XIL_RUN_EXIT_ERROR;



432                    }



433                    else {



434                        XIL_RUN_EXIT;



435                    }



436                }



437                break;



438            case XIL_PAUSE_COMMAND:



439                if (xilPause(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



440                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_INITIALIZE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



441                        XIL_RUN_EXIT_ERROR;



442                    }



443                    else {



444                        XIL_RUN_EXIT;



445                    }



446                }



447                break;



448            case XIL_SYSTEM_INITIALIZE_COMMAND:



449                if (xilSystemInitialize(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



450                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_SYSTEM_INITIALIZE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



451                        XIL_RUN_EXIT_ERROR;



452                    }



453                    else {



454                        XIL_RUN_EXIT;



455                    }



456                }



457                break;



458            case XIL_TERMINATE_COMMAND:



459                if (xilTerminate(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



460                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_TERMINATE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



461                        XIL_RUN_EXIT_ERROR;



462                    }



463                    else {



464                        XIL_RUN_EXIT;



465                    }



466                }



467                break;



468            case XIL_ENABLE_COMMAND:



469                if (xilEnable(cc.fcnId, cc.commandIdx) != XIL_INTERFACE_SUCCESS) {



470                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_ENABLE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



471                        XIL_RUN_EXIT_ERROR;



472                    }



473                    else {



474                        XIL_RUN_EXIT;



475                    }



476                }



477                break;



478            case XIL_DISABLE_COMMAND:



479                if (xilDisable(cc.fcnId, cc.commandIdx) != XIL_INTERFACE_SUCCESS) {



480                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_DISABLE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



481                        XIL_RUN_EXIT_ERROR;



482                    }



483                    else {



484                        XIL_RUN_EXIT;



485                    }



486                }



487                break;



488            case XIL_UPDATE_COMMAND:



489                if (xilUpdate(cc.fcnId, cc.commandIdx) != XIL_INTERFACE_SUCCESS) {



490                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_UPDATE, 1) != XIL_INTERFACE_LIB_SUCCESS) {



491                        XIL_RUN_EXIT_ERROR;



492                    } else {



493                        XIL_RUN_EXIT;



494                    }



495                }



496                break;



497#endif



498            case XIL_OUTPUT_COMMAND:



499                if (xilOutput(cc.fcnId, cc.commandIdx) != XIL_INTERFACE_SUCCESS) {



500                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_OUTPUT, 1) != XIL_INTERFACE_LIB_SUCCESS) {



501                        XIL_RUN_EXIT_ERROR;



502                    }



503                    else {



504                        XIL_RUN_EXIT;



505                    }



506                }



507                break;



508#ifndef OUT_OF_PROCESS_EXEC



509            case XIL_SYSTEM_RESET_COMMAND:



510                if (xilSystemReset(cc.fcnId) != XIL_INTERFACE_SUCCESS) {



511                    if (finalizeCommandResponse(XIL_RESPONSE_ERROR_SYSTEM_RESET, 1) != XIL_INTERFACE_LIB_SUCCESS) {



512                        XIL_RUN_EXIT_ERROR;



513                    }



514                    else {



515                        XIL_RUN_EXIT;



516                    }



517                }                     



518                break;



519#endif



520            default:



521                XIL_RUN_EXIT_ERROR;



522        }



523        /* compute next state */



524        cc.state = computeNextState(cc.state, cc.command);



525        XIL_EXIT_ERROR_IF_INVALID_STATE;



526    }



527    if (cc.state == XIL_PROCESS_TARGET_TO_HOST_DATA) {



528        XIL_PROCESSDATA_ERROR_CODE processDataError;



529        /* process data from target */



530        processDataError = processTargetToHostData(cc.fcnId, XIL_RESPONSE_OUTPUT_DATA, cc.command, cc.commandIdx);



531        if (processDataError == XIL_PROCESSDATA_DATA_STREAM_ERROR) {



532            XIL_RUN_EXIT_ERROR;



533        } else if (processDataError != XIL_PROCESSDATA_SUCCESS) {



534            if (finalizeCommandResponse(XIL_RESPONSE_ERROR_PROCESS_YDATA, 1) != XIL_INTERFACE_LIB_SUCCESS) {



535                XIL_RUN_EXIT_ERROR;



536            }



537            else {



538                XIL_RUN_EXIT;



539            }



540        }



541        /* compute next state */



542        cc.state = computeNextState(cc.state, cc.command);



543        XIL_EXIT_ERROR_IF_INVALID_STATE;



544        /*break;*/



545    }



546        



547



548#ifdef LDRA_ENABLED



549    /* upload code coverage execution history from target*/



550    if (cc.command == XIL_TERMINATE_COMMAND) {



551        callTargetUploadFcns();



552    }



553#endif /* LDRA_ENABLED */



554                



555    /* client/server commands could be executed in the middle of a step */



556    if ((cc.state == XIL_COMPLETE) && (cc.command != XIL_CLIENT_SERVER_COMMAND)) { 



557        /* finalize the response */



558        MemUnit_T isEndOfXILStep = 1;



559        if (finalizeCommandResponse(XIL_RESPONSE_ERROR_SUCCESS, isEndOfXILStep) != XIL_INTERFACE_LIB_SUCCESS) {



560            XIL_RUN_EXIT_ERROR;



561        }



562        



563        /* reset state */



564        cc.state = XIL_READ_COMMAND_CODE;



565    }



566



567    XIL_RUN_EXIT;



568    



569}

572uint8_T commsRun(void)



573{            



574    const uint8_T success = 1;



575



576    xilProcessMsg();



577    return success;

580XIL_INTERFACE_LIB_ERROR_CODE xilRun(void) {



581    XIL_INTERFACE_LIB_ERROR_CODE errorCode = XIL_INTERFACE_LIB_SUCCESS;



582    int success;             



583



584    /* read one msg and pass to owning application */



585    success = commsRun();



586    



587    success = success && processMsgSuccess;



588    if (success) {        



589        if (isShutdownCommand) {



590            errorCode = XIL_INTERFACE_LIB_TERMINATE;



591        } else {



592            errorCode = XIL_INTERFACE_LIB_SUCCESS;



593        }



594        



595    } else {



596        errorCode = XIL_INTERFACE_LIB_ERROR;



597    }



598    return errorCode;



599}

601/* Dispatch and receive client/server commands within a step */



602void xilCommandDispatchAndResponse(XIL_COMMAND_TYPE_ENUM xilCommandType, 



603                                   uint32_T xilCommandIdx, uint32_T xilFcnId, 



604                                   uint32_T svrFcnId, MemUnit_T xilResponseId,



605                                   XIL_PROCESS_MSG_CONTEXT_SWITCH callerCtx) {



606



607    /* define a binary flag, each CS command needs to process a response */



608    int needToProcessCSResponse = 1;



609    



610    /* update server function id */



611    serverFcnId = svrFcnId;        



612



613    /* send input data to the host */



614    if (processTargetToHostData(xilFcnId, xilResponseId, xilCommandType, 



615            xilCommandIdx) == XIL_PROCESSDATA_DATA_STREAM_ERROR) {



616        restoreProcessMsgContext(callerCtx);



617        XIL_RUN_EXIT_ERROR;



618    }



619    



620    /* finalize the response */



621    {



622        MemUnit_T isEndOfXILStep = 0;



623        if (finalizeCommandResponse(XIL_RESPONSE_ERROR_SUCCESS, isEndOfXILStep) !=



624                XIL_INTERFACE_LIB_SUCCESS) {



625            restoreProcessMsgContext(callerCtx);



626            XIL_RUN_EXIT_ERROR;



627        }



628    }



629    



630    /* reset state for new context, recursive calls are safe because all



631     * the caller contexts (callerCtx) are cached in the stack */



632    cc.state = XIL_READ_COMMAND_CODE;



633    



634    while(needToProcessCSResponse) {



635        /* read next command requested by the host */



636        xilReadCommandCode();



637        if (cc.command == XIL_CLIENT_SERVER_COMMAND) {



638            /* if the host sent a response to the server call, process it



639             * and then exit the loop. Otherwise, process the current 



640             * command and read the next one */



641            needToProcessCSResponse = 0;



642        }



643        /* process response data sent by host */



644        if (xilRun() != XIL_INTERFACE_LIB_SUCCESS) {



645            restoreProcessMsgContext(callerCtx);



646            XIL_RUN_EXIT_ERROR;



647        }



648    }



649    



650    restoreProcessMsgContext(callerCtx);



651    



652}

654XIL_PROCESS_MSG_CONTEXT_SWITCH xilGetCurrentProcessMsgContext(void) {



655    /* return the current context (cc) */



656    return cc;



657}

662void swapBytes(MemUnit_T * const buffer, uint32_T size) {



663    MemUnit_T tmp;



664    MemUnit_T *headPtr;



665    MemUnit_T *tailPtr;



666    uint32_T idx;



667



668    for(idx = 0; idx < size/2; idx ++) {



669        headPtr = buffer + idx; 



670        tailPtr = buffer + size -1 - idx;



671        tmp = *headPtr;



672        *headPtr = *tailPtr;



673        *tailPtr = tmp;



674    }



675}

677static XIL_DATA_STREAM_ERROR_CODE xilReadDataAndSwapBytes(MemUnit_T * dst, uint32_T size) {



678    XIL_DATA_STREAM_ERROR_CODE readDataVal = xilReadData(dst, size);



679



680    if (readDataVal == XIL_DATA_STREAM_SUCCESS) {



681        swapBytes(dst, size);



682    }



683    return readDataVal;



684}
